create table hotel(
	id bigint generated always as identity primary key,
	nome varchar(100) not null,
	estado varchar(40) not null,
	cidade varchar(40) not null,
	telefone varchar(50),
	andares integer,
	data_cadastro timestamp,
	data_atualizacao timestamp
);

create table reserva(
	id bigint generated always as identity primary key,
